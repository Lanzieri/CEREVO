    <!-- Script: ocultar header al scrollear hacia abajo, mostrar al subir -->
    <script is:inline>
      (() => {
        const el = document.querySelector("[data-header]");
        if (!el) return;

        let lastY = window.scrollY;
        const TH = 10; // umbral

        window.addEventListener("scroll", () => {
          const y = window.scrollY;
          if (y > lastY + TH) {
            el.classList.add("scrolled"); // se oculta (ver global.css)
          } else if (y < lastY - TH) {
            el.classList.remove("scrolled"); // se muestra
          }
          lastY = y;
        }, { passive: true });
      })();
    </script>
